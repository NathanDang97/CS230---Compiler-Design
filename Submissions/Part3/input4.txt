void main()
{
   int m, n;
   bool done;
   while (!done) {
      if (m % 2 == 0) {
         if (m < n) {
            m = m + 1;
            n = n - 1;
         }
         else {
            m = m - 1;
            n = n + 1;
         } 
      }
      else {
         if (m < n) {
            m = m + 2;
            n = n - 2;
         }
         else {
            m = m - 2;
            n = n + 2;
         }
      }
      if (m == n)
         done = !done;
   }
}
